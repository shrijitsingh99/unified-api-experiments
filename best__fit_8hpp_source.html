98 template <typename RuntimeChecks = executor_runtime_checks, typename Function,
99  typename... SupportedExecutors>
100 void enable_exec_with_priority(
101  Function&& f, std::tuple<SupportedExecutors...> supported_execs) {
102  static_assert(std::is_base_of<executor_runtime_checks, RuntimeChecks>::value,
103  "Runtime checks should inherit from executor_runtime_checks");
104  bool executor_selected = false;
105  executor::for_each_until_true(supported_execs, [&](auto& exec) {
106  if (RuntimeChecks::check(exec)) {
107  executor_selected = detail::execute(f, exec);
108  return executor_selected;
109  }
110  return false;
111  });
112 
113  // This should not happen, at least one executor should always be selected. So
114  // either all runtime checks failed which is incorrect behaviour or no
115  // supported executors were passed which should not be done
116  if (!executor_selected)
117  std::cerr << "No executor selected. All runtime checks returned false or "
118  "no executors were passed."
119  << std::endl;
120 }
